s(FVP,PS,TS) --> dp(FDP,PDP,TDP), vp(FVP,PVP,TVP), {    % determiner phrase + verb phrase
    att(FVP, subj, dp(FDP)),
    att(FDP, num, Num),     % checking for number agreement 
    att(FVP, num, Num),        % between subject and verb
    link(1, PVP, PDP, PS),
    TS = s(TDP,TVP) }.

    att(FV, cpl, [dp(FDP)]),
    link(2, PV,PDP,PVP),
    TVP = vp(TV,TDP) }.

vp(FV,PVP,TVP) --> v(FV,PV,TV), pp(FPP,PPP,TPP), pp(FPP2,PPP2,TPP2), { % ditransitive verb, eg. 'talk'
    att(FV, cpl, [pp(P1),pp(P2)]),
    att(FPP, gloss, P1),
    att(FPP2, gloss, P2),
    link(2, PV,PPP,PV1),
    link(3, PV1,PPP2,PVP),
    TVP = vp(TV, TPP, TPP2) }.
  SemanticAgreement
s(FVP,PS,TS) --> dp(FDP,PDP,TDP), vp(FVP,PVP,TVP), {    % determiner phrase + verb phrase
    att(FVP, subj, dp(FDP)),
    att(FDP, num, Num),     % checking for number agreement 
    att(FVP, num, Num),        % between subject and verb
    att(FVP, anim, A),
    att(FDP, anim, A),
    link(_, PVP, PDP, PS),
    TS = s(TDP,TVP) }.
